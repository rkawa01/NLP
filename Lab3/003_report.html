class RegexExtractor:
    """
    Uniwersalna klasa do .map(), ktra kompiluje wzorzec 
    i ekstrahuje wszystkie pene dopasowania do nowego pola.
    """
    def __init__(self, pattern_string, output_field_name, flags=0):
        import re
        
        self.pattern = re.compile(pattern_string, flags)
        self.output_field_name = output_field_name

    def __call__(self, example):
        matches = [match.group(0) for match in self.pattern.finditer(example['text'])]
        
        if matches:
            example[self.output_field_name] = list(set(matches)) # Unikalne
        else:
            example[self.output_field_name] = []
        return example

def run_llm_extraction(model_name: str, user_prompt: str, temperature: float = 0.0) -> ExtractionResponse:
    """Wywouje Ollama i zwraca obiekt Pydantic. Zwraca pusty w razie bdu."""
    SYSTEM_PROMPT = 'Jeste precyzyjnym, automatycznym ekstraktorem wzorcw. Zwracasz TYLKO i wycznie poprawny obiekt JSON w formacie {"matches": [...]}. Nie dodawaj adnych wyjanie ani markdowna. Bezwzgldnie przestrzegaj REGU i OGRANICZE podanych przez uytkownika. Jeli w tekcie ABSOLUTNIE nie ma adnych dopasowa, zwr pust list: {"matches": []}.'

    try:
        response = ollama.chat(
            model=model_name,
            messages=[
                {"role": "system", "content": SYSTEM_PROMPT},
                {"role": "user", "content": user_prompt}
            ],
            format=ExtractionResponse.model_json_schema(),
            options={
                'temperature': temperature, "repeat_penalty": 1.1
            }
        )
        json_string = response['message']['content']
        data = json.loads(json_string)
        validated_response = ExtractionResponse.model_validate(data)
        return validated_response
    except (json.JSONDecodeError, ValidationError, ollama.ResponseError, Exception) as e:
        #print(f"BD (model: {model_name}, pomijam): {e}") 
        return ExtractionResponse(matches=[])

def sample_by_text_length(dataset: Dataset, max_records: int = 100, max_text_length: int = 1000, num_proc: int = NUM_CORES_TO_USE) -> Dataset:
    """
    Zwraca nowy Dataset zawierajcy maksymalnie `max_records` prbek,
    ktrych pole 'text' ma dugo <= max_text_length.
    Jeli max_records jest None lub <= 0, zwraca wszystkie dopasowania.
    """
    if dataset is None:
        raise ValueError("dataset is None")

    def _keep(example):
        text = example.get("text", "")
        return len(text) <= int(max_text_length)

    filtered = dataset.filter(_keep, batched=False, num_proc=num_proc)

    if max_records is None or int(max_records) <= 0:
        return filtered

    n = min(int(max_records), len(filtered))
    return filtered.select(range(n))

def sample_by_text_length_truncate(dataset: Dataset, max_records: int = 100, max_text_length: int = 1000, num_proc: int = NUM_CORES_TO_USE) -> Dataset:
    """
    Zwraca Dataset z maksymalnie `max_records` prbkami.
    Teksty dusze ni max_text_length zostan przycite do tej dugoci
    (pole 'text' zastpowane jest wersj przycit).
    """
    if dataset is None:
        raise ValueError("dataset is None")

    def _truncate(example):
        text = example.get("text", "")
        if len(text) > int(max_text_length):
            example["text"] = text[: int(max_text_length)]
        return example

    truncated = dataset.map(_truncate, batched=False, num_proc=num_proc)

    if max_records is None or int(max_records) <= 0:
        return truncated

    n = min(int(max_records), len(truncated))
    return truncated.select(range(n))

import pandas as pd
def prf1(gold, pred):
    """Oblicza P/R/F1 dla pojedynczego porwnania."""
    G, P = set(gold), set(pred)
    tp = len(G & P)
    prec = tp / len(P) if P else 0
    rec  = tp / len(G) if G else 0
    f1   = 2 * prec * rec / (prec + rec) if (prec + rec) else 0
    return prec, rec, f1

def calculate_micro_metrics(dataset, gold_col, pred_col):
    """
    Oblicza metryki micro-average dla caego zbioru danych.
    Jest to bardziej miarodajne ni urednianie 500 wynikw F1.
    """
    total_tp = 0
    total_gold_len = 0
    total_pred_len = 0

    for row in dataset:
        G = set(row[gold_col])

def calculate_micro_metrics_all(dataset, gold_col, pred_col):
    """
    Oblicza sumy TP, FP, FN dla caego zbioru danych.
    """
    total_tp = 0
    total_fp = 0

    for row in dataset:
        gold_list = row[gold_col] if row[gold_col] is not None else []
        pred_list = row[pred_col] if row[pred_col] is not None else []

Odpowiedzi na pytania

  1. Polskie znaki i fleksja: Modele (Llama/Gemma) poprawnie rozpoznaway sowa z diakrytykami (np. zotych). Problem lea w precyzji: RegEx znajdowa tylko te formy, ktre mu podalimy, podczas gdy Llama 3.1, mimo rozumienia fleksji, generowaa przy tym tyle szumu (FP), e jej niska precyzja (5%) czynia j bezuyteczn. Ciko oceni czy w zadaniu z fleksj modele faktycznie co znalazy, poniewa naprawde sporo tam halucynoway, czsto po prostu przepisyway formy czowiek/ludzie z prompta zamiast faktycznie wyapywa wzorce z tekstu.

  2. Deterministyczno: temperature=0 jest kluczowe. W testach rzeczywicie zdarzao si, e halucynowao na poziomie 99% znajdowanych (JAKICHKOLWIEK) wzorcw gdzie wcale tyle nie byo (w niektrych zadaniach), aczkolwiek nie uwaam, e to naprawde poprawiao sytuacj, poniewa i tak precyzja bya tragiczna. Repeat_penalty=1.2 pomagao unika powtrze w generowanym wyjciu, aczkolwiek dugo tekstu jest kluczowa, eby si model nie zacina. Ale ostatecznie temperature=0 zmniejszya halucynacje wraz z wymuszeniem schematu JSON.

  3. Precyzja vs. Uoglnienie:

    RegEx pokaza wysok precyzj (P=~0.8), ale niski/redni recall (R=~0.4-0.6). By precyzyjny, ale gubi rzadkie przypadki (False Negatives), np. nie znalaz "1,5 do 3 tys. z".

    Llama 3.1 8b uoglniaa zbyt mocno (FP - False Positives), co dao katastrofaln precyzj (P=~0.05). Model generowa mas halucynacji, a jego Recall wcale nie by znaczco lepszy od RegEx.

  4. Odporno na szum: LLM teoretycznie rozumie literwki (500 zl vs z), ale w praktyce wprowadzi znacznie wicej "szumu" (halucynacji) ni rozwiza problemw. Ciekawe jest to, e gemma2:2b generowa mniej szumu ni llama 3.1 8b. Nie uwzgldniem go w ostatecznych testach, ale byo to widoczne w eksperymentach.

  5. Skalowanie z dugoci: To by krytyczny problem. Mniejsze modele (gemma2:2b) cakowicie zawieszay si na dugich tekstach (>1k znakw), oczywicie nie kady tekst, ale dugo 1k znakw bya bezpiecznym limitem. Llama 3.1 8b bya stabilniejsza (cho rwnie si zacinaa w niektrych przypadkach), ale jej skonno do halucynacji rosa wraz z dugoci tekstu.

  6. Formatowanie wyjcia (JSON): Llama 3.1 8b dziaaa idealnie (100% poprawnych JSON-w) dziki format=ExtractionResponse.model_json_schema(), temperature=0 i promptom XML. Gemma 2 2b jak i pozniej Llama 3.1 8b okazaa si niestabilna, poniewa si zacinaa bez odpowiednich ustawie (prompt + temperature=0 + repeat_penalty=1.2) oraz <OUTPUT> tagw w promptach oraz dugoci tekstu wejciowego.

  7. Transfer midzy kategoriami: Transfer by bardzo atwy. Struktura promptu XML (<RULES>) dziaaa troche jak szablon. Zmiana zadania z IBAN na Kwoty PLN wymagaa jedynie zmiany tekstu w <RULES>. Model nie powinien mc myli kategorii, poniewa prompty byy niezalenie puszczane dla kadej kategorii.

  8. Llama vs. Gemma (Jako/Prdko):

    Gemma 2b (mniejszy model) bya cakowicie niestabilna i bezuyteczna; zawieszaa si na niektrych danych dlatego te nie uwzgldniem jej w ostatecznych testach. Chocia moe warto byo zweryfikowa recall tego modelu dla testu.

    Llama 3.1 8b (wikszy model) bya stabilniejsza, ale cierpiaa na katastrofaln precyzj (5%) w zadaniach ekstrakcji.

Podsumowanie

Co bym wybra ?

  • RegExp: Do zada ekstracji danych o wyskoiej precyzji (IBAN, e-mail, daty cyfrowe, PLN)
  • LLM: Model nie okaza si przydatny do tego zadania z powodu precyzji najzwyczaniej. Mgby by uywany do prostych zada jak kategoryzacja tematw itd.

Czyli ostatecznie RegExp'a

Natomiast widz potencja w poczeniu obu technik, gdzie RegExp wyapuje to co potrafi z wysok precyzj, albo to co nam si wydaje jak poprawne na pierwszy rzut oka, a LLM mgby by uyty do znalezienia elastycznych formatw ktre RegExp pomin.

Rozbiym to na 3 kroki:

  1. Uzycie RegEx do ekstrakcji wzorcw "twardych" z wysok precyzj, do ktrych regex jest dobrze dostosowany.
  2. Uycie LLM do ekstrakcji wzorcw "mikkich" lub elastycznych formatw, ktre RegEx pomin.
  3. Walidacja wynikw LLM, uwzgldnienie brakujcych wzorcw w RegEx i ewentualne poprawki RegEx na podstawie wynikw LLM.

S jeszcze przykady gdzie RegEx moe si nie nadawa a LLM ju tak, np. ekstrakcja wzorcw sownych takich jak jeden milion zotych itp. gdzie RegEx nie jest w stanie sobie poradzi, a LLM mgby to zrobi lepiej. Wtedy dodatkowym krokiem byoby:

  1. Uycie LLM do ekstrakcji wzorcw sownych (np. kwoty sowne) i integracja ich z wynikami RegEx i LLM z krokw 1-3. Niestety to trzebaby robi pewnie rcznie, chyba e udaoby si to zautomatyzowa w jaki sposb.
